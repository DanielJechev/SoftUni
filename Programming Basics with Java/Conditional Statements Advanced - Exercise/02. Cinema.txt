package com.company;
import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String type = scanner.nextLine();
        int redove = Integer.parseInt(scanner.nextLine());
        int koloni = Integer.parseInt(scanner.nextLine());
         switch ( type) {
             case "Premiere":
                 double cena =12.00 ;
                 double prixodi = redove* koloni* cena;
                 System.out.printf("%.2f leva", prixodi);
                 break;
             case "Normal":
                 double cena1 =7.50;
                 double prixodi1 = redove* koloni* cena1;
                 System.out.printf("%.2f leva", prixodi1);
                 break;
             case "Discount":
                 double cena2=5.00;
                 double prixodi2 = redove* koloni* cena2;
                 System.out.printf("%.2f leva", prixodi2);
                 break;
                 default:
                     
                     break;





         }


    }
}
