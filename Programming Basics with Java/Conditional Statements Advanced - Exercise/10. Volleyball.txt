package com.company;

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);
        String godina = scanner.nextLine();
        int broiPrazniciVgodinata= Integer.parseInt(scanner.nextLine());
        int yikendiDoRodniqGrad=Integer.parseInt(scanner.nextLine());
        int yikend=48;
        int yikendSofiq= yikend-yikendiDoRodniqGrad;
        double igriVsofiq = yikendSofiq * 0.75  ;
        double igriPopraznici = broiPrazniciVgodinata * 0.6667;
        double sum = 0;
        if("leap".equals(godina)) {

            double sum1 = igriVsofiq + yikendiDoRodniqGrad + igriPopraznici;
            sum= sum1+ (0.15*sum1);
            System.out.printf("%.0f", Math.floor(sum));

        }else  if ("normal".equals(godina)){
            sum = igriVsofiq + yikendiDoRodniqGrad + igriPopraznici;
            System.out.printf("%.0f",Math.floor(sum));





        }
    }
}
