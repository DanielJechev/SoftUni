package com.company;

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
       int age =Integer.parseInt(scanner.nextLine());
       double pricemachine= Double.parseDouble(scanner.nextLine());
       int pricetoy=Integer.parseInt(scanner.nextLine());
       int savedMoney=0;
       int toysCount=0;


        for (int i = 1; i <=age; i++) {
            if (i % 2 == 0) {
                savedMoney+=i*5;
                savedMoney--;


            } else {
                toysCount ++;
            }

        }
        int totalSum=pricetoy*toysCount+ savedMoney;

        if (totalSum >= pricemachine ){
            System.out.printf("Yes! %.2f",totalSum-pricemachine);
        } else{
            System.out.printf("No! %.2f",pricemachine-totalSum);
        }
    }
    }
