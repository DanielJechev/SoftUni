package com.company;

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
       String name =scanner.nextLine();
       int num = name.length();
       int sum =0;
        for (int i= 0; i <num ; i++) {
            char symbol = name.charAt(i);
             switch (symbol){
                 case 'a':
                     sum=sum+1;
                     break;
                 case 'e':
                     sum=sum+2;
                     break;
                 case'i':
                     sum=sum+3;
                     break;
                 case'o':
                     sum=sum+4;
                     break;
                 case'u':
                     sum=sum+5;
                     break;
                     default:
                         break;

             }
        }
        System.out.println(sum);
    }
    }
